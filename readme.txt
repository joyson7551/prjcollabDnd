VITE_API_BASE_URL=http://localhost:YOUR-PORT

VITE_AUTH_BASE=/auth/user
VITE_PROJECT_BASE=/auth/project

# Backend
MONGO_URI=mongodb+srv://YOUR-MONGO_URI/projectcollaboration
PORT=YOUR-PORT
JWT_SECRET=YOUR_JWT_SCECRET
